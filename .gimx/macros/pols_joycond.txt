# for joycond
# 16, 17 -> 18, 19
#
# left
#  - input 16
#  - output 18
#  - alias F3
#  - for TRIGGER F1
# right
#  - input 17
#  - output 19
#  - alias F4
#  - for TRIGGER F2

MACRO JBUTTONDOWN 16
KEYDOWN F1
KEYDOWN F3
KEYUP F1

MACRO JBUTTONUP 16
KEYDOWN F1
KEYUP F3
KEYUP F1

MACRO JBUTTONDOWN 17
KEYDOWN F2
KEYDOWN F4
KEYUP F2

MACRO JBUTTONUP 17
KEYDOWN F2
KEYUP F4
KEYUP F2

# -------------------------------------

MACRO KEYDOWN F3
TRIGGER KEYUP F2
TOGGLE YES
INIT ON
JBUTTONDOWN 18

MACRO KEYDOWN F3
TRIGGER KEYDOWN F2
TOGGLE YES
JBUTTONDOWN 18
JBUTTONUP 19

MACRO KEYDOWN F4
TRIGGER KEYUP F1
TOGGLE YES
INIT ON
JBUTTONDOWN 19

MACRO KEYDOWN F4
TRIGGER KEYDOWN F1
TOGGLE YES
JBUTTONDOWN 19
JBUTTONUP 18

MACRO KEYUP F3
TRIGGER KEYUP F2
TOGGLE YES
INIT ON
JBUTTONUP 18

MACRO KEYUP F3
TRIGGER KEYDOWN F2
TOGGLE YES
JBUTTONUP 18
JBUTTONDOWN 19

MACRO KEYUP F4
TRIGGER KEYUP F1
TOGGLE YES
INIT ON
JBUTTONUP 19

MACRO KEYUP F4
TRIGGER KEYDOWN F1
TOGGLE YES
JBUTTONUP 19
JBUTTONDOWN 18
